## Ref

### refreshNotesV2

```ts

refreshNotesV2(notes) {
    notes.map {
        uri, id := it

        if it.status = 'create' {
            @history.add(action: create, uri)
        }
        @notes[id] = it
    }
    @fuseEngine.update(notes)
}

```
